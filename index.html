⚙ Configurações · Sincronização
O dashboard sincroniza automaticamente ao abrir e a cada 5 minutos enquanto estiver aberto.
Para funcionar em qualquer dispositivo: cole a URL do Apps Script abaixo e salve o arquivo HTML com essa URL já preenchida.
📋 Código do Apps Script (clique para ver) 
function doGet(e) {
  var SHEETS = {
    'Samu':     '1RIj9z-c0e6NWdSaX2SevN-jDoJruPknoBqTC2J-KKBo',
    'Angie':    '1tfPDgsKygQy_cE_RKfjALUBqZtbVTOj-Tbwl_C6vC5c',
    'Gabriela': '1Rhk8pcvTFxtOduGFEgBpA3GtR8V1Lh92kDDFYg1mEY0',
    'Flavia':   '1Y2m6-2l-DEnZzxJ3iE7v1033tE02jldkjKVwycm6RK8',
    'Gabes':    '1XXdbvagMUoCjHjAZmJD9iz6Rh9XtvrV18WEzYWDzUPM',
    'Patricia': '1ZNDyNV4JyooZCnKqeloop_GNZGGmzR0ZF011BRZqaCs',
    'Eliza':    '1GgmPfRHaVoS9EiSLt8kvMUBDgDk2upi9LY9sHQEgsNo',
    'Laura':    '1XNHgIKalR7-IDDWia8WpQEmbCUlr-Er2AR4ugvBoBg4',
    'Juliana':  '1EW5niOgx5VAZSqlb46v-UlzRl3ewTjsmIhD5xknA0eE',
    'Alesson':  '1oqVLjEERQk9VVvK7G7pz4OlQHfE3NmqOCQf0M9ZUVMI'
  };
  var result = { teachers:{}, updated:new Date().toISOString() };
  for (var t in SHEETS) {
    try {
      var ss = SpreadsheetApp.openById(SHEETS[t]);
      var td = {};
      ss.getSheets().forEach(function(sh){
        td[sh.getName()] = sh.getDataRange().getValues();
      });
      result.teachers[t] = td;
    } catch(e){ result.teachers[t]={error:e.toString()}; }
  }
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
Como publicar o Apps Script:
1. Acesse script.google.com → Novo projeto
2. Cole o código acima e salve
